select SRONAM.NANAME, 
SRONAM.NATREG, 
SRONAM.NAADR2, 
SRONAM.NAADR3, 
SRONAM.NASTAT, 
SRONAM.NANUM, 
SROCMA.CMCONT, 
SRONFP.NFPHNO, 
SRONFP.NFCTX2, 
SRONFP.NFDFFX, 
SRONFP.NFDEID,
SRONAM.NADENO 
from SROCMA 
    inner join SRONAM on  SRONAM.NANUM=SROCMA.CMCUNO 
    left join SRONFP on SRONAM.NANUM=SRONFP.NFNANO
left join SRONAD on SRONAD.ADNUM=SRONAM.NANUM
where SRONAM.NATYPP in(1,3) 
    and SRONAM.NACRDT >= '<$0$>'
    and (SRONAM.NATREG='<$1$>' or '<$1$>'='')
order by SRONAM.NANUM
